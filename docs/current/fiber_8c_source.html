80 static fbr_id_t fbr_id_pack(struct fbr_fiber *fiber)
81 {
82  return ((__uint128_t)fiber->id << 64) | (uint64_t)fiber;
83 }

85 static int fbr_id_unpack(FBR_P_ struct fbr_fiber **ptr, fbr_id_t id)
86 {
87  struct fbr_fiber *fiber;
88  uint64_t f_id;
89  f_id = (uint64_t)(id >> 64);
90  fiber = (struct fbr_fiber *)(uint64_t)id;
91  if (fiber->id != f_id)
92  return_error(-1, FBR_ENOFIBER);
93  if (ptr)
94  *ptr = fiber;
95  return 0;
96 }

98 static void pending_async_cb(EV_P_ ev_async *w, _unused_ int revents)
99 {
100  struct fbr_context *fctx;
101  struct fbr_id_tailq_i *item;
102  fctx = (struct fbr_context *)w->data;
103  int retval;
104 
105  ENSURE_ROOT_FIBER;
106 
107  if (TAILQ_EMPTY(&fctx->__p->pending_fibers)) {
108  ev_async_stop(EV_A_ &fctx->__p->pending_async);
109  return;
110  }
111 
112  item = TAILQ_FIRST(&fctx->__p->pending_fibers);
113  assert(item->head == &fctx->__p->pending_fibers);
114  /* item shall be removed from the queue by a destructor, which shall be
115  * set by the procedure demanding delayed execution. Destructor
116  * guarantees removal upon the reclaim of fiber. */
117  ev_async_send(EV_A_ &fctx->__p->pending_async);
118 
119  retval = fbr_transfer(FBR_A_ item->id);
120  if (-1 == retval && FBR_ENOFIBER != fctx->f_errno) {
121  fbr_log_e(FBR_A_ "libevfibers: unexpected error trying to call"
122  " a fiber by id: %s",
123  fbr_strerror(FBR_A_ fctx->f_errno));
124  }
125 }

367 static void fbr_free_in_fiber(_unused_ FBR_P_ _unused_ struct fbr_fiber *fiber,
368  void *ptr, int destructor)
369 {
370  struct mem_pool *pool_entry = NULL;
371  if (NULL == ptr)
372  return;
373  pool_entry = (struct mem_pool *)ptr - 1;
374  if (pool_entry->ptr != pool_entry) {
375  fbr_log_e(FBR_A_ "libevfibers: address %p does not look like "
376  "fiber memory pool entry", ptr);
377  if (!RUNNING_ON_VALGRIND)
378  abort();
379  }
380  LIST_REMOVE(pool_entry, entries);
381  if (destructor && pool_entry->destructor)
382  pool_entry->destructor(FBR_A_ ptr, pool_entry->destructor_context);
383  free(pool_entry);
384 }

479 static enum ev_action_hint prepare_ev(FBR_P_ struct fbr_ev_base *ev)
480 {
481  struct fbr_ev_watcher *e_watcher;
482  struct fbr_ev_mutex *e_mutex;
483  struct fbr_ev_cond_var *e_cond;
484  struct fbr_id_tailq_i *item = &ev->item;
485 
486  ev->arrived = 0;
487  ev->item.dtor.func = item_dtor;
488  ev->item.dtor.arg = item;
489  fbr_destructor_add(FBR_A_ &ev->item.dtor);
490 
491  switch (ev->type) {
492  case FBR_EV_WATCHER:
493  e_watcher = fbr_ev_upcast(ev, fbr_ev_watcher);
494  if (!ev_is_active(e_watcher->w)) {
495  fbr_destructor_remove(FBR_A_ &ev->item.dtor,
496  0 /* call it */);
497  return EV_AH_EINVAL;
498  }
499  e_watcher->w->data = e_watcher;
500  ev_set_cb(e_watcher->w, ev_watcher_cb);
501  break;
502  case FBR_EV_MUTEX:
503  e_mutex = fbr_ev_upcast(ev, fbr_ev_mutex);
504  if (0 == e_mutex->mutex->locked_by) {
505  e_mutex->mutex->locked_by = CURRENT_FIBER_ID;
506  return EV_AH_ARRIVED;
507  }
508  id_tailq_i_set(FBR_A_ item, CURRENT_FIBER);
509  item->ev = ev;
510  ev->data = item;
511  TAILQ_INSERT_TAIL(&e_mutex->mutex->pending, item, entries);
512  item->head = &e_mutex->mutex->pending;
513  break;
514  case FBR_EV_COND_VAR:
515  e_cond = fbr_ev_upcast(ev, fbr_ev_cond_var);
516  if (0 == e_cond->mutex->locked_by) {
517  fbr_destructor_remove(FBR_A_ &ev->item.dtor,
518  0 /* call it */);
519  return EV_AH_EINVAL;
520  }
521  id_tailq_i_set(FBR_A_ item, CURRENT_FIBER);
522  item->ev = ev;
523  ev->data = item;
524  TAILQ_INSERT_TAIL(&e_cond->cond->waiting, item, entries);
525  item->head = &e_cond->cond->waiting;
526  fbr_mutex_unlock(FBR_A_ e_cond->mutex);
527  break;
528  }
529  return EV_AH_OK;
530 }

552 int fbr_ev_wait(FBR_P_ struct fbr_ev_base *events[])
553 {
554  struct fbr_fiber *fiber = CURRENT_FIBER;
555  struct fbr_ev_base **ev_pptr;
556  enum ev_action_hint hint;
557  int num = 0;
558 
559  fiber->ev.arrived = 0;
560  fiber->ev.waiting = events;
561 
562  for (ev_pptr = events; NULL != *ev_pptr; ev_pptr++) {
563  hint = prepare_ev(FBR_A_ *ev_pptr);
564  switch (hint) {
565  case EV_AH_OK:
566  break;
567  case EV_AH_ARRIVED:
568  fiber->ev.arrived = 1;
569  (*ev_pptr)->arrived = 1;
570  break;
571  case EV_AH_EINVAL:
572  return_error(-1, FBR_EINVAL);
573  }
574  }
575 
576  while (0 == fiber->ev.arrived)
577  fbr_yield(FBR_A);
578 
579  for (ev_pptr = events; NULL != *ev_pptr; ev_pptr++) {
580  if ((*ev_pptr)->arrived) {
581  num++;
582  finish_ev(FBR_A_ *ev_pptr);
583  } else
584  cancel_ev(FBR_A_ *ev_pptr);
585  }
586  return_success(num);
587 }

624 int fbr_fd_nonblock(FBR_P_ int fd)
625 {
626  int flags, s;
627 
628  flags = fcntl(fd, F_GETFL, 0);
629  if (flags == -1)
630  return_error(-1, FBR_ESYSTEM);
631 
632  flags |= O_NONBLOCK;

959 fbr_id_t fbr_create(FBR_P_ const char *name, fbr_fiber_func_t func, void *arg,
960  size_t stack_size)
961 {
962  struct fbr_fiber *fiber;
963  if (!LIST_EMPTY(&fctx->__p->reclaimed)) {
964  fiber = LIST_FIRST(&fctx->__p->reclaimed);
965  LIST_REMOVE(fiber, entries.reclaimed);
966  } else {
967  fiber = malloc(sizeof(struct fbr_fiber));
968  memset(fiber, 0x00, sizeof(struct fbr_fiber));
969  if (0 == stack_size)
970  stack_size = FBR_STACK_SIZE;
971  stack_size = round_up_to_page_size(stack_size);
972  fiber->stack = mmap(NULL, stack_size, PROT_READ | PROT_WRITE,
973  MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
974  if (MAP_FAILED == fiber->stack)
975  err(EXIT_FAILURE, "mmap failed");
976  fiber->stack_size = stack_size;
977  (void)VALGRIND_STACK_REGISTER(fiber->stack, fiber->stack +
978  stack_size);
979  fiber->id = fctx->__p->last_id++;
980  }
981  coro_create(&fiber->ctx, (coro_func)call_wrapper, FBR_A, fiber->stack,
982  fiber->stack_size);
983  fiber->call_list = NULL;
984  fiber->call_list_size = 0;
985  LIST_INIT(&fiber->children);
986  LIST_INIT(&fiber->pool);
987  TAILQ_INIT(&fiber->destructors);
988  strncpy(fiber->name, name, FBR_MAX_FIBER_NAME);
989  fiber->func = func;
990  fiber->func_arg = arg;
991  LIST_INSERT_HEAD(&CURRENT_FIBER->children, fiber, entries.children);
992  fiber->parent = CURRENT_FIBER;
993  return fbr_id_pack(fiber);
994 }
